#### 回溯

- 外部顺序为数字的顺序，如先从2 "abc"选，再从3 "def"选
- 内部顺序为字母选择的顺序，对于2的外部循环来说，他是从0到abc.size() - 1

```c++
class Solution {
public:
    string temp;
    vector<string> ans;
    //start是指先选择pick中的哪一个数字（由于答案可以按任意顺序返回），所以顺序是可以随意设置的，我们让他按照从0开始的顺序进行选择
    void backtrack(vector<string>& pick, int start){
        //如果temp和pick一样长，则此时已经选完，插入ans
        if(temp.size() == pick.size()){
            ans.push_back(temp);
            return;
        }

        //第一选从abc中选，先选a，再选b，最后选c
        for(int i = 0; i < pick[start].size(); i++){
            temp += pick[start][i];
            //第一步走完，第一步是从pick[start + 1]，即"def"中选
            backtrack(pick, start + 1);
            temp.pop_back();
        }
    }

    vector<string> letterCombinations(string digits) {
        vector<string> vec = {"", "", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"};
        vector<string> pick;
        //pick里面装了digits对应的字符串["abc", "def"]
        for(char digit : digits){
            pick.push_back(vec[digit - '0']);
        }
        backtrack(pick, 0);
        return ans;
    }
};
```
