#### 基础题

1. 在l1和l2都非空的情况下，循环插入
2. 在循环结束后，判断l1和l2哪个还有剩，把剩下的插到后面

```c++
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        ListNode* dummy = new ListNode(-1);
        ListNode* temp = dummy;
        ListNode* l1 = list1;
        ListNode* l2 = list2;

        while(l1 && l2){
            if(l1->val <= l2->val){
                temp->next = l1;
                temp = temp->next;
                l1 = l1->next;
            }
            else{
                temp->next = l2;
                temp = temp->next;
                l2 = l2->next;
            }
        }

        if(l1){
            temp->next = l1;
        }
        if(l2){
            temp->next = l2;
        }
        return dummy->next;
    }
};
```
