#### 特殊解法题

1. 把初始队列按照身高从低到高，如果身高一致则k从大到小排序（这个需要用sort加lambda才能写出来）
2. 排完序之后，遍历该队列，把每个人插在第k + 1个空位即可

```c++
class Solution {
public:
    vector<vector<int>> reconstructQueue(vector<vector<int>>& people) {
        int n = people.size();
        vector<vector<int>> ans(n);
        sort(people.begin(), people.end(), [](vector<int>& u, vector<int>& v){
            return u[0] < v[0] || (u[0] == v[0] && u[1] > v[1]);
        });

        for(vector<int>& person : people){
            //拿到当前人的k + 1
            int space = person[1] + 1;
            //把他放到ans里面的第k + 1个空位
            for(int i = 0; i < n; i++){
                if(ans[i].empty()){
                    space--;
                }
                if(space == 0){
                    ans[i] = person;
                    break;
                }
            }
        }
        return ans;
    }
};
```
