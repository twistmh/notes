#### 动态规划

如何判断`dp[i][j]`是否为回文子串：

1. 如果i和j相等，则一定是回文子串
2. 如果`j == i + 1`，当`nums[i] == nums[j]`时，是回文子串
3. 其他情况下，当`nums[i] == nums[j] && dp[i + 1][j - 1] == true` 时，是回文子串

- 由于`dp[i][j]`依赖`dp[i + 1][j - 1]`，所以i需要从大到小遍历，而j需要从小到大遍历

```c++
class Solution {
public:
    string longestPalindrome(string s) {
        int n = s.size();
        
        //dp[i][j]代表以i为左边界j为有边界的子串是不是回文子串
        vector<vector<bool>> dp(n, vector<bool>(n, false));

        int head = -1;
        int maxLength = 0;

        for(int i = n - 1; i >= 0; i--){
            for(int j = i; j < n; j++){
                if(i == j){
                    dp[i][j] = true;
                    if(1 > maxLength){
                        maxLength = 1;
                        head = i;
                    }
                }
                else if(j == i + 1){
                    if(s[i] == s[j]){
                        dp[i][j] = true;
                        if(2 > maxLength){
                            maxLength = 2;
                            head = i;
                        }
                    }
                }
                else{
                    if(s[i] == s[j] && dp[i + 1][j - 1] == true){
                        dp[i][j] = true;
                        if(j - i + 1 > maxLength){
                            maxLength = j - i + 1;
                            head = i;
                        }
                    }
                }
            }
        }

        return s.substr(head, maxLength);
    }
};
```
