#### BFS与层次遍历

通过size来记录每一层的结点数，然后让所有结点出队列，再让他们的儿子进队列，就可以让队列总是包含一层的全部结点

```c++
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> ans;
        queue<TreeNode*> q;
        //先把队列的第一层，也就是根节点push
        q.push(root);
        if(root == NULL){
            return ans;
        }

        while(!q.empty()){
            //每一层遍历开始前，先计算这一层的结点数
            int size = q.size();
            vector<int> row;

            //将这一层的所有结点pop出来并存到row中，得到这一层的vector并加入到ans中；把pop出来的结点的左右儿子，也就是下一层的所有结点push进队列中
            for(int i = 0; i < size; i++){
                TreeNode* temp = q.front();
                q.pop();
                row.push_back(temp->val);

                if(temp->left){
                    q.push(temp->left);
                }
                if(temp->right){
                    q.push(temp->right);
                }
            }
            ans.push_back(row);
        }
        return ans;
    }
};
```
