#### 回溯

------

**1.用temp存储路径，用visited避免重复**

```c++
class Solution {
public:
    bool ans = false;

    void backtrack(vector<vector<char>>& board, string word, vector<bool>& visited, string& temp, int i, int j, int start){
        //路径走完的return
        if(temp == word){
            ans = true;
            return;
        }

        //路径走错的return
        if(i < 0 || i >= board.size() || j < 0 || j >= board[0].size() || board[i][j] != word[start] || visited[i * board[0].size() + j] == true){
            return;
        }

        temp += word[start];
        visited[i * board[0].size() + j] = true;
        backtrack(board, word, visited, temp, i - 1, j, start + 1);
        backtrack(board, word, visited, temp, i + 1, j, start + 1);
        backtrack(board, word, visited, temp, i, j - 1, start + 1);
        backtrack(board, word, visited, temp, i, j + 1, start + 1);
        visited[i * board[0].size() + j] = false;
        temp.pop_back();
    }

    bool exist(vector<vector<char>>& board, string word) {
        int m = board.size();
        int n = board[0].size();
        vector<bool> visited(m * n, false);
        string temp;

        //寻找word的开头元素
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                if(board[i][j] == word[0]){
                    backtrack(board, word, visited, temp, i, j, 0);
                }
            }
        }

        return ans;
    }
};
```

**2.通过直接修改board来避免重复，用index判断路劲是否等于word，用返回值来判断是否有路径等于word**

```c++
class Solution {
public:
    bool backtrack(vector<vector<char>>& board, string& word, int i, int j, int index){
        //和递归一样，当递归首次越界的时候，就说明递归到底了，这里说明路径已经走到了word的最后，也就是有路径等于word了
        if(index == word.size()) return true;

        //除了递归的越界返回之外，还有回溯的错误路径返回
        if(i < 0 || i >= board.size() || j < 0 || j >= board[0].size() || board[i][j] != word[index]){
            return false;
        }

        //通过直接修改已经访问的元素来避免他被重复访问，并不会影响到后续的backtrack，因为回溯最终会还原场景
        char temp = board[i][j];
        board[i][j] = '!';

        //只要有一条路径能成立，那么对于backtrack i j来讲，就是成立的
        bool ret = backtrack(board, word, i - 1, j, index + 1) ||
                    backtrack(board, word, i + 1, j, index + 1) ||
                    backtrack(board, word, i, j - 1, index + 1) ||
                    backtrack(board, word, i, j + 1, index + 1);
        
        board[i][j] = temp;

        return ret;
    }

    bool exist(vector<vector<char>>& board, string word) {
        int m = board.size();
        int n = board[0].size();

        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                bool b = backtrack(board, word, i, j, 0);
                if(b == true) return true;
            }
        }
        return false;
    }
};
```
