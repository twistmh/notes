#### 哈希表

把nums存入哈希表set，有两个作用：

1. 由于重复数字在这里只有干扰作用，所以set可以帮助自动剔除重复数字
2. set的查找效率极高，在确定一个序列的第一个数字后，就可以一个个把后面的所有数字查找出来

```c++
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        int n = nums.size();
        //把nums存入哈希表
        unordered_set<int> hash(nums.begin(), nums.end());
        int ans = 0;
        int length = 0;

        //遍历set中的元素，如果该元素-1在set中不存在，说明该元素是一个序列的开头
        for(int num : hash){
            if(hash.count(num - 1) == 0){
                int curr = num;
                length = 1;
                //逐个遍历序列后面的元素，直到他不在set中存在为止，这样就可以得到一个子序列
                while(hash.count(curr + 1)){
                    curr++;
                    length++;
                }
            }
            ans = max(ans, length);
        }
        return ans;
    }
};
```
