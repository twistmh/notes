#### 特殊数组转换为环形链表

```c++
class Solution {
public:
    int findDuplicate(vector<int>& nums) {
        //数组中的环形链表
        //对于数组来说nums[i] = j也类似于一种链式的关系，i→j
        //如果数组中存在两个i，那么两个i都会指向j，相当于在链表中形成了环
        //但是这种方法只对一种特殊的数组有用，就是j的范围不能超过i索引的范围，这道题就刚好符合
        int n = nums.size();
        
        int slow = 0;
        int fast = 0;

        //slow走一步，fast走两步
        do{
            slow = nums[slow];
            fast = nums[nums[fast]];
        }while(nums[slow] != nums[fast]);

        //slow和fast相遇之后，让slow回到起点
        slow = 0;

        //slow走一步，fast走一步，再次相遇的点就是环的起点
        while(nums[slow] != nums[fast]){
            slow = nums[slow];
            fast = nums[fast];
        }

        return nums[slow];
    }
};
```
