#### 动态规划

```c++
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int n = nums.size();
        vector<int> dp(n, INT_MIN);
        int maxSum = INT_MIN;

        for(int i = 0; i < n; i++){
            if(i == 0){
                dp[i] = nums[0];
            }
            else{
                dp[i] = max(dp[i - 1] + nums[i], nums[i]);
            }
            maxSum = max(maxSum, dp[i]);
        }

        return maxSum;
    }
};
```
