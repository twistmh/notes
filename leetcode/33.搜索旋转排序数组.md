#### 特殊的二分查找

- 首先，对旋转排序数组取mid，mid的左边和右边至少有一边是有序数组，无序的那边无法判断，所以只需要判断有序的这边是否包含target，不包含则直接无条件扔给另一边
- 所以判断的顺序则变成了：先判断是左有序还是右有序，然后再判断target是否在有序的那一边

```c++
class Solution {
public:
    int search(vector<int>& nums, int target) {
        int n = nums.size();
        int left = 0;
        int right = n - 1;
        int ans = -1;

        while(left <= right){
            int mid = (left + right) / 2;
            if(nums[mid] == target){
                ans = mid;
                break;
            }
            //如果左半边是有序的
            if(nums[0] <= nums[mid]){
                //如果target在左半边的范围内，则到左边找
                if(target >= nums[0] && target < nums[mid]){
                    right = mid - 1;
                }
                //否则到右边找
                else{
                    left = mid + 1;
                }
            }
            //如果左半边无序，那么右半边一定是有序的，至少有一边是有序的，甚至可能两边都有序
            else{
                if(target <= nums[n - 1] && target > nums[mid]){
                    left = mid + 1;
                }
                else{
                    right = mid - 1;
                }
            }
        }
        return ans;
    }
};
```
