窗口长度固定为p.size()

**如何判断两个窗口是否是异位词**

- 异位词的每个单词的数量是相等的，比如a都是2，b都是3
- 为了给每个字母计数，可以立刻想到map<char, int>
- 但是由于小写字母只有26个，并且可以用-'a'转换为数字，所以map就麻烦了，用vector更好

```c++
class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        //子串的长度是固定的3，用滑动窗口
        //问题在于怎么判断两个长度为3的字符串是否是异位词：常规情况下用map做计数是最合适的，但是由于小写字母只有26个，所以用一个长为26的数组来代表26个字母的计数要更加节省资源
        int n = p.size();

        if(s.size() < n) return {};
        vector<int> pCount(26, 0);
        vector<int> sCount(26, 0);

        vector<int> ans;
        //初始化第一个窗口，长度为n
        for(int i = 0; i < n; i++){
            sCount[s[i] - 'a']++;
            pCount[p[i] - 'a']++;
        }

        //判断第一个窗口是否满足条件
        if(sCount == pCount){
            ans.push_back(0);
        }

        //开始进行滑动窗口，窗口的有边界的起始点为n
        for(int i = n; i < s.size(); i++){
            sCount[s[i] - 'a']++;
            sCount[s[i - n] - 'a']--;

            if(sCount == pCount){
                ans.push_back(i - n + 1);
            }
        }

        return ans;
    }
};
```
