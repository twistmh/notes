#### 动态规划

- 看似很难实际简单的题

------

1. 初始化部分，对于`dp[0][j]`，word1的前0个变成word2的前j个，只需要j次添加；`dp[i][0]`同理
2. dp迭代部分，分为两种情况：
   1. word1[i - 1] == word2[j - 1]，由于最后的元素相等，所以只需要前面的元素相等，就不需要额外添加步骤，`dp[i][j] = dp[i - 1][j - 1]`
   2. 如果不相等，则分为三种情况：①word1添加一个元素等于word2，`dp[i][j - 1] + 1`；②word1修改一个元素等于word2，`dp[i - 1][j - 1] + 1`；③word1删除一个元素等于word2，`dp[i - 1][j] + 1`

```c++
class Solution {
public:
    int minDistance(string word1, string word2) {
        int Len1 = word1.size();
        int Len2 = word2.size();

        vector<vector<int>> dp(Len1 + 1, vector<int>(Len2 + 1));

        //word1的前i个变成word2的前0个，只需要删除i个元素，所以需要i步
        for(int i = 0; i <= Len1; i++){
            dp[i][0] = i;
        }

        //word1的前0个变成word2的前j个，只需要添加j个元素，所以需要j步
        for(int j = 0; j <= Len2; j++){
            dp[0][j] = j;
        }

        for(int i = 1; i <= Len1; i++){
            char digit1 = word1[i - 1];
            for(int j = 1; j <= Len2; j++){
                char digit2 = word2[j - 1];
                if(digit1 == digit2){
                    dp[i][j] = dp[i - 1][j - 1];
                }
                else{
                    dp[i][j] = min({dp[i][j - 1] + 1, dp[i - 1][j] + 1, dp[i - 1][j - 1] + 1});
                }
            }
        }
        return dp[Len1][Len2];
    }
};
```
