#### 特殊题

- 将intervals中的区间按照起始点从小到大排序
- 新建一个`vector<vector<int>> ans`，如果intervals中的区间比ans.back()的结束点还要大，那么说明两个区间没有交集，直接插入；如果不是，说明有交集，判断intervals区间的尾和ans.back()的尾哪个更大，更新为更大的那一个

```c++
class Solution {
public:
    vector<vector<int>> merge(vector<vector<int>>& intervals) {
        int n = intervals.size();
        //将intervals中的元素按照起始点从小到大排序
        sort(intervals.begin(), intervals.end());

        vector<vector<int>> ans;

        for(int i = 0; i < n; i++){
            if(i == 0){
                ans.push_back(intervals[0]);
            }
            else{
                //如果intervals的区间的头比ans中区间的尾还大，那就不重叠，直接插进来
                if(intervals[i][0] > ans.back()[1]){
                    ans.push_back(intervals[i]);
                }
                else{
                    if(intervals[i][1] > ans.back()[1]){
                        ans.back()[1] = intervals[i][1];
                    }
                }
            }
        }

        return ans;
    }
};
```
