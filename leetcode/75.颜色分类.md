#### 特殊题

```c++
class Solution {
public:
    void sortColors(vector<int>& nums) {
        int p0 = 0;
        int p2 = nums.size() - 1;
        int curr = 0;

        while(curr <= p2){
            //p2可能是0，如果p2是0，那么交换后，curr不能往前走，而是要再和p0交换
            if(nums[curr] == 2){
                swap(nums[curr], nums[p2]);
                p2--;
            }
            //为什么跟p0交换可以直接让curr++？因为curr是从p0开始往右走的，p0在curr的前面，而curr的前面不可能还有2，因为2早就在curr在前面走的时候被换掉了
            else if(nums[curr] == 0){
                swap(nums[curr], nums[p0]);
                p0++;
                curr++;
            }
            else{
                curr++;
            }
        }
    }
};
```
