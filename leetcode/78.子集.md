#### 回溯

- 子集不能重复，也就是说1，2，3和3，2，1不能同时出现，这意味着不同顺序，但是相同元素的路径会被判为重复，也就是说路径只能朝一个方向前进，不能出现倒着走的路径；我们选择nums索引从小到大为正方向，那么在路径中选元素的时候，就不能往nums的左边选，也就是需要剔除左边的元素
- 子集内部的元素不能重复，也就是说在选路径的过程中，已经选过的元素不能再选，所以在路径前进的过程中，需要剔除已经选过的元素

```c++
class Solution {
public:
    vector<vector<int>> ans;
    vector<int> sub;

    void backtrack(vector<int>& nums, int length, int start){
        if(sub.size() == length){
            ans.push_back(sub);
            return;
        }

        //nums中的所有元素都需要做一次路径的头，并且随着for循环向后，前面的元素要被剔除，因为子集不能重复
        for(int i = start; i < nums.size(); i++){
            sub.push_back(nums[i]);
            //路径中已经选过的元素不能选第二次，因为子集内部的元素也不能重复
            backtrack(nums, length, i + 1);
            sub.pop_back();
        }
    }

    vector<vector<int>> subsets(vector<int>& nums) {
        ans.push_back(sub);

        int n = nums.size();
        for(int length = 1; length <= n; length++){
            backtrack(nums, length, 0);
        }
        return ans;
    }
};
```
