#### 先排序，再用双指针

- 这道题的恶心的地方几乎全部在去重
- 我们先选择第一个元素，然后再选择第二个和第三个元素，第一个元素不能重复，每一个第一个元素对应的第二个和第三个元素也不能重复；也就是需要去两次重

```c++
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> answer;
        //给nums排序
        sort(nums.begin(), nums.end());
        //如果第一个元素比0大或者最后一个元素比0小，则直接返回
        if(nums.front() > 0 || nums.back() < 0){
            return answer;
        }
        int n = nums.size();
        for(int i = 0; i < n; i++)
        {
            //由于三元组不能重复，所以要跳过重复的元素；这里是让第一个元素不重复
            if(i > 0 && nums[i] == nums[i-1]){
                continue;
            }
            int l = i + 1;
            int r = n - 1;
            //和二分的条件不一样，这里l和r是不能相等的，因为是一个三元组
            while(r > l)
            {
                int sum = nums[i] + nums[l] + nums[r];
                if(sum == 0){
                    answer.push_back({nums[i], nums[l], nums[r]});
                    //跳过重复的元素，这里是让第二个和第三个元素不重复
                    while(nums[l] == nums[l + 1] && l < r -1)  l++;
                    while(nums[r] == nums[r - 1] && l < r - 1)  r--;
                    l++;
                    r--;
                }
                else if(sum < 0){
                    l++;
                }
                else{
                    r--;
                }
            }
        }
        return answer;
    }
};
```
