#### 回溯

- 要输出所有的组合，而不是只要组合数，所以是回溯不是动态规划
- 元素可以被重复选取，但是一个组合只能输出一种顺序，[2,2,3]和[2,3,2]不能都输出，所以在选完第一个元素后，进入第二个元素循环时，第一个元素就不应该继续在选择范围内，用start控制每一轮的起始位置，排除前面的元素

```c++
class Solution {
public:
    vector<int> temp;
    int sum = 0;
    vector<vector<int>> ans;

    void backtrack(vector<int>& candidates, int target, int start){
        if(sum == target){
            ans.push_back(temp);
            return;
        }
        if(sum > target){
            return;
        }

        for(int i = start; i < candidates.size(); i++){
            temp.push_back(candidates[i]);
            sum += candidates[i];
            //当路径的第一个为candidates[1]时，candidates[0]就必须被排除，否则会出现重复
            backtrack(candidates, target, i);
            sum -= candidates[i];
            temp.pop_back();
        }
    }

    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        backtrack(candidates, target, 0);
        return ans;
    }
};
```
