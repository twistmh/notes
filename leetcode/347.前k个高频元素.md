#### 小顶堆

如果把次数一个个插入小顶堆，并且保持小顶堆的大小为k，那么最后留下来的k个次数就是最大的k个次数

1. 统计nums中所有数字出现的次数，用map记录
2. 创建一个小顶堆
3. 将次数作为排序依据插入到小顶堆里面，最后剩下的k个次数就是频率最高的

```c++
class Solution {
public:

    vector<int> topKFrequent(vector<int>& nums, int k) {
        //<数字, 次数>
        unordered_map<int, int> counts;

        for(int num : nums){
            counts[num]++;
        }

        //创建小顶堆，默认是大顶堆，加上greater是小顶堆，依照pair的first进行排序
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;

        for(pair<int, int> count : counts){
            pq.push({count.second, count.first});
            if(pq.size() > k){
                pq.pop();
            }
        }

        vector<int> ans;
        while(!pq.empty()){
            ans.push_back(pq.top().second);
            pq.pop();
        }
        return ans;
    }
};

```
