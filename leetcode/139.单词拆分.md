#### 动态规划

s的前i个字母能不能拆，与前面的所有i - 1个状态都有关，所以每次计算dp[i]时，都需要遍历前面的所有状态



```c++
class Solution {
public:
    bool wordBreak(string s, vector<string>& wordDict) {
        int n = s.size();
        //set的查找性能强
        unordered_set<string> words(wordDict.begin(), wordDict.end());
        //dp[i]表示s的前i个字符组成的字符串是否可以被拆分
        vector<bool> dp(n + 1);
        //当i=0，默认可以被拆分
        dp[0] = true;

        //对于dp[i]，如果dp[j] == true且(j, i)的部分存在于字典中，那么dp[i]就可分
        for(int i = 1; i <= n; i++){
            for(int j = 0; j <= i; j++){
                if(dp[j] == true && words.count(s.substr(j, i - j))){
                    dp[i] = true;
                    break;
                }
            }
        }
        return dp[n];
        
    }
};
```
