#### 动态规划

类似于打家劫舍二叉树版，我们需要记录多种状态的迭代

```c++
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n = prices.size();

        //在某一时间点，总共可能有三种状态，我们先将第一天的状态初始化
        int p0 = -prices[0];//持有
        int p1 = 0;//不持有，可买
        int p2 = 0;//不持有，不可买

        for(int i = 1; i < n; i++){
            int new_p0 = max(p0, p1 - prices[i]);//持有状态可能由持有和不持有可买两种状态变化而来
            int new_p1 = max(p1, p2);//不持有可买状态可能由不持有可买状态和不持有不可买状态变化而来
            int new_p2 = p0 + prices[i];//不持有不可买状态只可能由持有状态变化而来

            p0 = new_p0;
            p1 = new_p1;
            p2 = new_p2;
        }
        return max(p1, p2);
    }
};
```
