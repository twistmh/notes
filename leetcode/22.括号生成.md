#### 回溯

- 回溯的不同路径之间不是if,else的关系，而是都要走一遍的关系

```c++
class Solution {
public:
    void backtrack(vector<string>& ans, int n, string& s, int left, int right){
        if(left + right == n*2){
            ans.push_back(s);
            return;
        }
		
        //先左后右，虽然有if判定，但是每个路径都是需要走的
        if(left < n){
            s.push_back('(');
            backtrack(ans, n, s, left + 1, right);
            s.pop_back();
        }
        if(right < left){
            s.push_back(')');
            backtrack(ans, n, s, left, right + 1);
            s.pop_back();
        }

    }

    vector<string> generateParenthesis(int n) {
        vector<string> ans;
        string s = "";
        backtrack(ans, n, s, 0, 0);
        return ans;
    }
};
```
