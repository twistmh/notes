#### 回溯

- 这道题的回溯条件只有“索引不重复”，没有起始点等等别的条件
- 为了记录索引是否已经访问过，需要一个visited数组

```c++
class Solution {
public:
    void backtrack(vector<int>& nums, vector<int>& temp,
                   vector<vector<int>>& ans, int n, vector<bool>& visited) {
        if (temp.size() == n) {
            ans.push_back(temp);
            return;
        }

        for (int i = 0; i < n; i++) {
            if (visited[i] == false) {
                temp.push_back(nums[i]);
                visited[i] = true;
                backtrack(nums, temp, ans, n, visited);
                visited[i] = false;
                temp.pop_back();
            }
        }
    }

    vector<vector<int>> permute(vector<int>& nums) {
        int n = nums.size();
        vector<int> temp;
        vector<vector<int>> ans;
        vector<bool> visited(n, false);
        backtrack(nums, temp, ans, n, visited);
        return ans;
    }
};
```
