#### 贪心

```c++
class Solution {
public:
    bool canJump(vector<int>& nums) {
        int n = nums.size();
        int farest = 0;

        //贪心思想，动态地记录当前可以跳到的最远点，如果连nums[i]都跳不到，那自然不可能跳到更后面的地方，如果已经可以跳到nums[n - 1]了，那就说明一定可以跳到，不再考虑其他情况
        for(int i = 0; i < n; i++){
            if(farest >= n - 1) return true;
            if(farest < i) return false;
            farest = max(farest, i + nums[i]);
        }
        //如果上面的for一次都没return，只有一种可能，就是nums只有一个元素
        return true;
    }
};
```
