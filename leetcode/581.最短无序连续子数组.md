#### 特殊题

```c++
class Solution {
public:
    int findUnsortedSubarray(vector<int>& nums) {
        //左边界和有边界初始为无效值，如果最后还是无效值，说明没有进行任何操作，也就是说nums本身就是有序的
        int leftBound = -1;
        int rightBound = -1;
        //让maxval为最小值，保证第一次比较一定可以赋值
        int maxVal = INT_MIN;
        int minVal = INT_MAX;

        int n = nums.size();

        //从左往右遍历，维护一个最大值，如果当前元素比最大值小，就把右边界挪到这个元素
        for(int i = 0; i < n; i++){
            maxVal = max(maxVal, nums[i]);
            if(nums[i] < maxVal){
                rightBound = i;
            }
        }

        //从右往左遍历，维护一个最小值，如果当前元素比最小值大，就把左边界挪到这个元素
        for(int j = n - 1; j >= 0; j--){
            minVal = min(minVal, nums[j]);
            if(nums[j] > minVal){
                leftBound = j;
            }
        }

        if(leftBound == -1) return 0;
        return rightBound - leftBound + 1;
    }
};
```
