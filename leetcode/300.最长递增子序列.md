#### 动态规划

让dp[i]表示以i号元素为结尾的最长递增子序列的长度

- 初始化状态为i = 0，把ans也初始化为这种状态

```c++
class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n = nums.size();
        //dp[i]代表以i为结尾的最长递增子序列的长度，dp[i]保底就是1，所以初始化为1
        vector<int> dp(n, 1);
        if(n == 1) return 1;
        
        //把ans初始化为只有一个元素的情况
        int ans = 1;

        for(int i = 1; i < n; i++){
            for(int j = 0; j < i; j++){
                if(nums[j] < nums[i]){
                    dp[i] = max(dp[i], dp[j] + 1);
                }
            }
            ans = max(ans, dp[i]);
        }
        return ans;
        
    }
};
```
