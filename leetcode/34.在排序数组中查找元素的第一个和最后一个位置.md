#### 特殊的二分

一般情况下，二分只要nums[mid] == target就会得到结果，但是这里不一样，如果要找第一个，那么在nums[mid] == target时还要继续查找左半部分，如果要找最后一个，那么在nums[mid] == target时还要继续查找右半部分



- 可以把两种情况写在一起，但是比较复杂；用两次二分分别查找第一个和最后一个比较直观

```c++
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int n = nums.size();
        int left = 0;
        int right = n - 1;
        int first = -1;
        int last = -1;

        //先找左边界
        while(left <= right){
            int mid = (left + right) / 2;
            if(nums[mid] == target){
                first = mid;
                right = mid - 1;
            }
            else if(nums[mid] > target){
                right = mid - 1;
            }
            else{
                left = mid + 1;
            }
        }

        //再找右边界
        left = 0;
        right = n - 1;
        while(left <= right){
            int mid = (left + right) / 2;
            if(nums[mid] == target){
                last = mid;
                left = mid + 1;
            }
            else if(nums[mid] > target){
                right = mid - 1;
            }
            else{
                left = mid + 1;
            }
        }
        vector<int> ans = {first, last};
        return ans;
    }
};
```
