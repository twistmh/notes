#### 栈

纯粹折磨



思路：

- 用栈来存储s的不同元素

1. 如果s[i]是数字，把以这个数字开头的整个数字拿出来，插入到栈里面
2. 如果是'['或者字母，直接插入
3. 如果是']'，反向遍历把两个括号中间的字母拿出来组成字符串，再把前面的数字拿出来当倍率，然后把指定倍率的字符串再插入到栈里面



其中涉及很多复杂操作：

1. 将多个char类型的数字转换为一个string数字 '1', '2'----"12"
2. 将多个string类型的字母转换为一个string字符串    "a", "b", "c" -----"abc"
3. 判断一个字符是否为字母isalpha，判断一个字符是否为数字isdigit
4. 将数字字符串变成数字stoi

```c++
class Solution {
public:
    vector<string> stk;

    //把数作为一个字符串拿出来
    string getNumber(string s, int& i, int n) {
        string ret = "";
        while(isdigit(s[i])){
            ret += s[i];
            i++;
        }
        return ret;
    }

    string getWord(vector<string>& word){
        string ret = "";
        for(string s : word){
            ret += s;
        }
        return ret;
    }

    string decodeString(string s) {
        int i = 0;
        int n = s.size();
        while(i != n) {
            // 如果是数字，就把整个数字字符串拿出来插入stk，i会在getNumber里面到达下一个非数字的位置
            if (isdigit(s[i])) {
                string num = getNumber(s, i, n);
                stk.push_back(num);
            }
            //如果是字母或者是[，直接把他变成字符串插入到stk里面即可
            else if(isalpha(s[i]) || s[i] == '['){
                string temp(1, s[i]);
                stk.push_back(temp);
                i++;
            }
            //如果是]
            else{
                vector<string> word;
                //把字母拿出来存到word里面，是["c", "b", "a"]的形式
                while(stk.back() != "["){
                    word.push_back(stk.back());
                    stk.pop_back();
                }
                //把"["弹出
                stk.pop_back();

                //把word变成正的["a", "b", "c"]
                reverse(word.begin(), word.end());

                //sub是"abc"
                string sub = getWord(word);

                //取前面的数字
                int times = stoi(stk.back());
                stk.pop_back();

                for(int j = 0; j < times; j++){
                    stk.push_back(sub);
                }

                i++;
            }
        }
        return getWord(stk);
    }
};
```
