#### 动态规划与0-1背包

- 只要能在nums里面抽出一批数加起来等于总和的一半，那么就可以满足条件
- 背包就是前i个元素，选或不选其中的元素，使其和为sum的一半

```c++
class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int n = nums.size();

        int sum = 0;
        for(int num : nums){
            sum += num;
        }
        //如果nums的总和是奇数，那么直接返回false
        if(sum % 2 != 0) return false;

        //只要有一批数加起来等于sum/2，那么另一批肯定也是sum/2
        int target = sum / 2;
        //前i个元素里面能否凑出一批元素加起来等于target

        vector<vector<bool>> dp(n + 1, vector<bool>(target + 1, false));

        //无论是前多少个元素，一定可以凑出0（什么都不选）
        for(int i = 0; i <= n; i++){
            dp[i][0] = true;
        }

        for(int i = 1; i <= n; i++){
            //在i = 1的这个循环，边界数为nums[i - 1]
            int num = nums[i - 1];
            for(int j = 0; j <= target; j++){
                if(j >= num){
                    dp[i][j] = dp[i - 1][j] || dp[i - 1][j - num];
                }
                else{
                    dp[i][j] = dp[i - 1][j];
                }
            }
        }
        return dp[n][target];
    }
};
```

------

由于第i行只与第i-1行有关，所以也可以优化到一维

```c++
class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int n = nums.size();

        int sum = 0;
        for(int num : nums){
            sum += num;
        }
        //如果nums的总和是奇数，那么直接返回false
        if(sum % 2 != 0) return false;

        //只要有一批数加起来等于sum/2，那么另一批肯定也是sum/2
        int target = sum / 2;
        //前i个元素里面能否凑出一批元素加起来等于target

        vector<bool> dp(target + 1, false);

        //无论是前多少个元素，一定可以凑出0（什么都不选）
        dp[0] = true;

        for(int num : nums){
            for(int j = target; j >= num; j--){
                    dp[j] = dp[j] || dp[j - num];
            }
        }
        return dp[target];
    }
};
```
